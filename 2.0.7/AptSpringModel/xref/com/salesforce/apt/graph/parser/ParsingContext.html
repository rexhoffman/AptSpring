27  package com.salesforce.apt.graph.parser;
28  
29  import java.util.ArrayList;
30  import java.util.List;
31  import java.util.Queue;
32  
33  import com.salesforce.apt.graph.model.DefinitionModel;
34  import com.salesforce.apt.graph.model.errors.ErrorMessages;
35  import com.salesforce.apt.graph.model.errors.ErrorModel;
36  import com.salesforce.apt.graph.model.storage.DefinitionModelStore;
37  import com.salesforce.apt.graph.processing.Verifier;
38  import com.salesforce.apt.graph.types.AssignabilityUtils;
39  
40  public class ParsingContext {
41  
42    private final ErrorMessages messageFormats;
43  
44    protected final List<DefinitionModel> definitions = new ArrayList<>();
45      
46    protected final AssignabilityUtils assignabilityUtils;
47    
48    protected final DefinitionModelStore store;
49    
50    public ErrorMessages getMessageFormats() {
51      return messageFormats;
52    }
53    
54    public ParsingContext(ErrorMessages messageFormats,
55        AssignabilityUtils assignabilityUtils, DefinitionModelStore store) {
56      this.messageFormats = messageFormats;
57      this.assignabilityUtils = assignabilityUtils;
58      this.store = store;
59    }
60  
61    public void addDefinition(DefinitionModel definition) {
62      this.definitions.add(definition);
63    }
64    
65    public Queue<ErrorModel> checkAndStoreValid() {
66      return new Verifier().verifyDefinitions(this.definitions, assignabilityUtils, store);
67    }
68    
69    /**
70     * Processors seem to be reused across successful builds in eclipse m2e-apt.
71     * To prevent internal state from causing issues, wipe that state by calling this
72     * method when the final round of processing is complete.
73     */
74    public void reset() {
75      definitions.clear();
76    }
77  }
