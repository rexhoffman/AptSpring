73    /**
74     * The the sha 256 of dependencies against the stored data.
75     * 
76     * @param model model who's dependencies we will inspect
77     * @param store the store of all the model data (abstraction, will likely be in memory or class files)
78     * @param errorListener if any shas mismatch will report here.
79     * @return true if all shas match.
80     */
81    public boolean verifiedShas(DefinitionModel model, DefinitionModelStore store, Consumer<ErrorModel> errorListener) {
82      boolean verified = true;
83      for (DefinitionModel dep : model.getDependencies()) {
84        if (!dep.getSourceElement().isPresent()  //not recompiling
85            && model.getDependencyNameToSha256().containsKey(dep.getIdentity())) { //model already has a sha256 of it
86          if (!model.getDependencyNameToSha256().get(dep.getIdentity()).equals(dep.getSha256())) {
87            errorListener.accept(new ErrorModel(ErrorType.DEPENDENCY_SHA_MISMATCH,
88                  Arrays.asList(model, dep),  Arrays.asList(model)));
89            verified = false;
90          }
91        }
92      }
93      return verified;
94    }

97    /**
98     * Verify that all expected entities are marked expected.
99     * Verify that the types of provided entities satisfy all expected types....
100    * <p/>
101    * if a child record was freshly process, so too must all parent ( if the md5 has changed ).
102    *  
103    * @param definition the head of current tree of definitions linked by imports
104    * @param errorListener registers all errors found 
105    */
106   private boolean depthFirstExpectedsInspector(DefinitionModel definition, Consumer<ErrorModel> errorListener,
107       AssignabilityUtils assignabilityUtils, DefinitionModelStore store) {
108     boolean errored = false;
109     for (DefinitionModel dependency : definition.getDependencies()) {
110       if (!dependency.isLockedAnalyzed()) {
111         errored = depthFirstExpectedsInspector(dependency, errorListener, assignabilityUtils, store) || errored;
112         if (errored) {
113           return true; // no need to continue.
114         }
115       } else {
116         if (!verifiedShas(dependency, store, errorListener)) {
117           errored = true;
118         }
119       }
120     }
121     
122     if (errored) {
123       return errored;
124     }
125     
126     //check that each object has only one source ( could be the imported from a dependency, in a diamond pattern )
127     //validate the expected beans are correct.
128     Map<String, InstanceModel> resolvedInstances = ensureSingleInstanceOfEachName(definition, errorListener);
129     
130     //short circuit if dependencies couldn't be resolved.
131     if (resolvedInstances == null) {
132       return false;
133     }
134     
135     //looks for cycles and unexpected missing entities.
136     errored = detectCyclesInEntityGraph(definition, resolvedInstances, errorListener);
137 
138     //check types of non-expected dependencies
139     errored = checkInstancesTypesInDefinition(definition, resolvedInstances, errorListener, assignabilityUtils) || errored;
140 
141     //check all definitions with expected, that each instance expecting a definition can use the supplied.
142     errored = checkProvidedSupplyCorrectTypes(definition, resolvedInstances, errorListener, assignabilityUtils) || errored;
143 
144     //prune all edged from graph that don't end in an expected
145 
146     //store computed expects with all types that must satisfied.
147     
148     
149     
150     if (!errored) {
151       //store as provided dependencies
152       definition.addAllProvidedInstances(resolvedInstances.values());
153       for (DefinitionModel dep : definition.getDependencies()) {
154         definition.addDependencyNameToSha256(dep.getIdentity(), dep.getSha256());
155       }
156       //storing will lock a the definition, as will reading.
157       if (!store.store(definition)) {
158         errorListener.accept(
159             new ErrorModel(ErrorType.COULD_NOT_STORE, Arrays.asList(definition), Arrays.asList(definition)));
160       }
161     }
162     
163     return errored;
164   }

166   private ErrorModel errorForMismatchedExpected(final DefinitionModel definition, ExpectedModel computedExpected,
167       final Map<String, InstanceModel> nameToEntity, AssignabilityUtils assignabilityUtils) {
168     //that which is to fill all the expectedInstance references.
169     InstanceModel providedInstance = nameToEntity.get(computedExpected.getIdentity());
170     
171     //the top level definition model should have declared this an expectedBean - maybe move that error here?
172     if (providedInstance == null) {
173       return null;
174     }
175     
176     //Instances which expect to use the provided instance
177     List<InstanceModel> mistmatchedExpectantEntities = computedExpected.getDefinitionReferenceToType().keySet().stream()
178         .map(name -> nameToEntity.get(name))
179         .filter(expectantInstance -> !assignabilityUtils.isAssignableFrom(providedInstance, expectantInstance))
180         .collect(Collectors.toList());
181     
182     if (mistmatchedExpectantEntities.size() == 0) {
183       return null;
184     } else {
185       List<InstanceModel> involved = new ArrayList<>();
186       involved.add(providedInstance);
187       involved.addAll(mistmatchedExpectantEntities);
188       return new ErrorModel(ErrorType.UNMATCHED_TYPES, involved, Arrays.asList(definition, providedInstance));
189     }
190   }

192   private boolean checkProvidedSupplyCorrectTypes(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
193       final Consumer<ErrorModel> errorListner, AssignabilityUtils assignabilityUtils) {
194     List<ErrorModel> errors = definition.getDependencies().stream()
195         .flatMap(dependency -> dependency.getComputedExpected().stream())
196         .filter(expectedModel -> !definition.getExpectedDefinitions().stream()
197             .anyMatch(ed -> ed.getIdentity().equals(expectedModel.getIdentity())))
198         .map(em -> errorForMismatchedExpected(definition, em, nameToEntity, assignabilityUtils))
199         .filter(errorModel -> errorModel != null)
200         .collect(Collectors.toList());
201     errors.stream().forEach(errorListner);
202     return errors.size() > 0;
203   }

206   private List<Entry<String, InstanceModel>> getEntryListForNameAndAlias(InstanceModel instanceModel) {
207     List<Entry<String, InstanceModel>> output = new ArrayList<>();
208     output.add(new SimpleEntry<>(instanceModel.getIdentity(), instanceModel));
209     for (String alias : instanceModel.getAliases()) {
210       if (!alias.equals(instanceModel.getIdentity())) {
211         output.add(new SimpleEntry<>(alias, instanceModel));
212       }
213     }
214     return output;
215   }

217   /**
218    * Verify that a single named instance model is correctly identifiable composed of owningDefinition/ElementLocation/Identity.
219    * 
220    * 
221    * @param definition the definition of 
222    * @param errorListener accepts and displays all errors produced by analyzing the models.
223    * @return returns a map of each instance name to the single model that the has been resolved to. 
224    */
225   private Map<String, InstanceModel> ensureSingleInstanceOfEachName(DefinitionModel definition,
226       Consumer<ErrorModel> errorListener) {
227     Map<String, Map<String, InstanceModel>> instancesByNameAndLocationDedupped = 
228         definitionToAllInstancesByNameAndSourceLocation(definition);
229     
230     final Map<String, InstanceModel> resolvedDependencies = new HashMap<>();
231     boolean errored = false;    
232     for (Entry<String, Map<String, InstanceModel>> entry : instancesByNameAndLocationDedupped.entrySet()) {
233       if (entry.getValue().size() == 1) {
234         resolvedDependencies.put(entry.getKey(), entry.getValue().values().iterator().next()); //get only InstanceModel.
235       } else {
236         errored = true;
237         errorListener.accept(errorForDuplicateInstanceModels(definition, entry.getValue().values().stream()
238             .sorted((i1, i2) -> i1.getElementLocation().compareTo(i2.getElementLocation()))
239             .collect(Collectors.toList())));
240       }
241     }
242     return errored ? null : resolvedDependencies;
243   }

245   /**
246    * Give a definition model, extract all instances from this definition, and all imported definitions.
247    * Group the instances by name, and then map them from source location to InstanceModel.
248    *  
249    * @param definition model to extract all instance information from, including imported definitions.
250    * @return a map of maps, [name -> [sourceLocation -> instanceModel]]
251    */
252   private Map<String, Map<String, InstanceModel>> definitionToAllInstancesByNameAndSourceLocation(
253       DefinitionModel definition) {
254     //create a stream of all imported Definition's InstanceModels
255     Stream<InstanceModel> imported = definition.getDependencies().stream()
256         .map(d -> d.getProvidedInstances()).flatMap(x -> x.stream());
257     //merge the stream with all local instance models.
258     Stream<InstanceModel> instanceModelStream = Stream.concat(definition.getObjectDefinitions().stream(), imported);
259 
260     //map all instances by name and then by source location (due to diamond dependencies in definition imports.
261     Map<String, Map<String, InstanceModel>> instancesByNameAndLocationDedupped = instanceModelStream
262         .flatMap(instance -> getEntryListForNameAndAlias(instance).stream()) //flat map all alias to entries
263         .collect(Collectors.groupingBy(entry -> entry.getKey(), //group by name
264             Collectors.mapping(entry -> entry.getValue(), 
265                 Collectors.toMap(im -> im.getElementLocation(), //by location
266                     im -> im, //identity function for first insert
267                     //choose the one with source element on merge, if any
268                     (im1, im2) -> im1.getSourceElement().isPresent() ? im1 : im2))));
269     return instancesByNameAndLocationDedupped;
270   }

272   /**
273    * Given a list of duplicate instance models (same identifier, different source elements) produce a 
274    * well formed ErrorModel.
275    * 
276    * @param definition where the error was first detected.
277    * @param causes the list of all instances models from the definition or any of it's imported definitions.
278    * @return an well formed ErrorModel of the duplicated beans.
279    */
280   private ErrorModel errorForDuplicateInstanceModels(DefinitionModel definition, List<InstanceModel> causes) {
281     ArrayList<AbstractModel> involved = new ArrayList<>();
282     involved.add(definition);
283     involved.addAll(causes.stream()
284         .filter(instanceModel -> instanceModel.getSourceElement().isPresent())
285         .collect(Collectors.toList()));
286     return new ErrorModel(ErrorType.DUPLICATE_OBJECT_DEFINITIONS, causes, involved);
287   }

289   /**
290    * Inspects the instance graph for cycles, any cycle is printed as an error.   The nameToEntity parameter doesn't list expected
291    * instances, any instances that are not found in the nameToInstances map (they are looked for because they are referenced as a
292    * dependency by an instance in the map) and are not found by name in the definition's expectedInstances are treated as errors
293    * as well.
294    * 
295    * @param definition definition being processed.  Will uses it's expected list, any instances references as dependencies but
296    *     not found, not listed as expected in this DefinitionModel, will be treated as errors.
297    * @param nameToEntity name to unique instanceModels, verified before call.
298    * @param errorListener accepts and displays all errors produced by analyzing the models
299    * @return true if an error occurred, false otherwise
300    */
301   private boolean detectCyclesInEntityGraph(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
302       final Consumer<ErrorModel> errorListener) {
303     final Map<String, ExpectedModel> missing = new HashMap<>();
304     final Graph<BaseInstanceModel, DefaultEdge> entityGraph = new DefaultDirectedGraph<>(DefaultEdge.class);
305     for (BaseInstanceModel entity : nameToEntity.values()) {
306       if (!entityGraph.containsVertex(entity)) {
307         entityGraph.addVertex(entity);
308       }
309       if (InstanceModel.class.isAssignableFrom(entity.getClass())) {
310         InstanceModel/com/salesforce/apt/graph/model/InstanceModel.html#InstanceModel">InstanceModel instanceModel = (InstanceModel) entity;
311         for (InstanceDependencyModel instanceDependency : instanceModel.getDependencies()) {
312           BaseInstanceModel dependency = nameToEntity.get(instanceDependency.getIdentity());
313           if (dependency == null) {
314             dependency = missing.computeIfAbsent(instanceDependency.getIdentity(), s -> new ExpectedModel(s));
315             missing.get(instanceDependency.getIdentity())
316               .addDefinitionReferenceToType(instanceModel.getIdentity(), instanceDependency.getType());
317           }
318           if (!entityGraph.containsVertex(dependency)) {
319             entityGraph.addVertex(dependency);
320           }
321           entityGraph.addEdge(entity, dependency);
322         }
323       }
324     }
325     
326     boolean errored = errorsForCycles(errorListener, entityGraph);
327     errored = testAllMissingEntitiesAreExpected(definition, errorListener, missing, entityGraph) || errored;
328     errored = errorUnusedExpectedsOnDefinition(definition, errorListener, missing) || errored;
329     return errored;
330   }

332   private boolean checkInstancesTypesInDefinition(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
333       final Consumer<ErrorModel> errorListner, AssignabilityUtils assignabilityUtils) {
334     boolean errored = false;
335     List<InstanceModel> instances = definition.getObjectDefinitions();
336     for (InstanceModel instance : instances) {
337       for (InstanceDependencyModel instanceDependency : instance.getDependencies()) {
338         InstanceModel dependency = nameToEntity.get(instanceDependency.getIdentity());
339         if (dependency != null && !assignabilityUtils.isAssignableFrom(dependency, instance)) {
340           errored = true;
341           errorListner.accept(new ErrorModel(ErrorType.UNMATCHED_TYPES,
342               Arrays.asList(instance, dependency), Arrays.asList(instance, dependency)));
343         }
344       }
345     }
346     return errored;
347   }

349   private boolean errorUnusedExpectedsOnDefinition(final DefinitionModel definition, final Consumer<ErrorModel> errorListner,
350       final Map<String, ExpectedModel> missing) {
351     boolean errored = false;
352     for (ExpectedModel expected : definition.getExpectedDefinitions()) {
353       if (!missing.keySet().contains(expected.getIdentity())) {
354         errorListner.accept(new ErrorModel(ErrorType.UNUSED_EXPECTED, Arrays.asList(expected), Arrays.asList(definition)));
355         errored = true;
356       }
357     }
358     return errored;
359   }

361   private boolean errorsForCycles(final Consumer<ErrorModel> errorListner,
362       final Graph<BaseInstanceModel, DefaultEdge> entityGraph) {
363     SzwarcfiterLauerSimpleCycles<BaseInstanceModel, DefaultEdge> cycleFind = new SzwarcfiterLauerSimpleCycles<>();
364     boolean errored = false;
365     cycleFind.setGraph(entityGraph);
366     List<List<BaseInstanceModel>> cycles = cycleFind.findSimpleCycles();
367     for (List<BaseInstanceModel> cycle : cycles) {
368       errored = true;
369       errorListner.accept(new ErrorModel(ErrorType.CYCLE_IN_DEFINITION_SOURCES, cycle, cycle));
370     }
371     return errored;
372   }

374   private boolean testAllMissingEntitiesAreExpected(final DefinitionModel definition, final Consumer<ErrorModel> errorListner,
375       final Map<String, ExpectedModel> missing, final Graph<BaseInstanceModel, DefaultEdge> entityGraph) {
376     //check computed expected are actually expected
377     boolean errored = false;
378     List<String> expectedMissing = definition.getExpectedDefinitions().stream().map(em -> em.getIdentity())
379         .collect(Collectors.toList());
380     for (ExpectedModel expected : missing.values()) {
381       if (!expectedMissing.contains(expected.getIdentity())) {
382         List<AbstractModel> dependsOnMissing = Stream.concat(
383             Stream.of(definition), 
384             entityGraph.incomingEdgesOf(expected).stream().map(edge -> entityGraph.getEdgeSource(edge))
385               .filter(m -> m.getSourceElement().isPresent()))
386             .collect(Collectors.toList());
387         errored = true;
388         errorListner.accept(new ErrorModel(ErrorType.MISSING_BEAN_DEFINITIONS, Arrays.asList(expected), dependsOnMissing));
389       } else {
390         definition.addComputedExpected(expected);
391       }
392     }
393     return errored;
394   }
