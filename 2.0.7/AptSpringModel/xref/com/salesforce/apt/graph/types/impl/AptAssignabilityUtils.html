54     * if the subject can be assigned to the dependency on the subject, from the
55     * target.
56     *
57     * @param subject instance to be injected in to the target
58     * @param target instance to have the subject injected in to it.
59     * @return whether the target can be safely injected in to the correct parameter of the target.
60     */
61    public boolean isAssignableFrom(InstanceModel../../../../com/salesforce/apt/graph/model/InstanceModel.html#InstanceModel">InstanceModel subject, InstanceModel target) {
62      ExecutableElement factoryOrConstructor = lookUpElement(subject);
63      TypeMirror subjectElementType = null;
64      if ("<init>".equals(factoryOrConstructor.getSimpleName().toString())) {
65        subjectElementType = factoryOrConstructor.getEnclosingElement().asType();
66      } else {
67        subjectElementType = factoryOrConstructor.getReturnType();
68      }
69      int count = 0;
70      while (!target.getDependencies().get(count).getIdentity().equals(subject.getIdentity())) {
71        count ++;
72      }
73      TypeMirror targetElementType = lookUpElement(target).getParameters().get(count).asType();
74      return typeUtils.isAssignable(subjectElementType, targetElementType);
75    }
