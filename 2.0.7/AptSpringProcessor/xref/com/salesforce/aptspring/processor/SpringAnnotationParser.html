112   /**
113    * Will return true if a class level contains exactly a constant final static private literal field.
114    */
115   private Predicate<VariableElement> staticPrivateFinalLiteralField = ve -> ve.getModifiers()
116       .containsAll(Arrays.asList(Modifier.PRIVATE, Modifier.STATIC, Modifier.FINAL))
117       && ve.getModifiers().size() == 3
118       && ve.getConstantValue() != null;
119 
120   /**
121    * Will return true if a class level contains exactly a final private field without a constant value.
122    */
123   private Predicate<VariableElement> privateFinalField = ve -> ve.getModifiers()
124       .containsAll(Arrays.asList(Modifier.PRIVATE, Modifier.FINAL))
125       && ve.getModifiers().size() == 2
126       && ve.getConstantValue() == null;

128   /**
129    * Used to construct entries for maps.
130    * @param key entry's key
131    * @param value entry's value
132    * @return the SimpleEntry we want to construct.
133    */
134   private static <K, V> Map.Entry<K, V> entry(K key, V value) {
135     return new AbstractMap.SimpleEntry<>(key, value);
136   }
137   
138   /**
139    * Converts a Stream&lt;Entry&lt;X,Y&gt;&gt; to a Map&lt;X,Y&gt;.
140    * @return a Map representing the contents of the stream.
141    */
142   private static <K, U> Collector<Map.Entry<K, U>, ?, Map<K, U>> entriesToMap() {
143     return Collectors.toMap((e) -> e.getKey(), (e) -> e.getValue());
144   }

158   /**
159    * Read a TypeElement to get application structure.
160    * 
161    * @param te definition type element.
162    * @param messager presents error messages for the compiler to pass to the user.
163    * @return the {@link DefinitionModel} parsed from a properly annotated {@link TypeElement}
164    */
165   public DefinitionModel extractDefinition(TypeElement te, Messager messager) {  
166     Verified verified = te.getAnnotation(Verified.class);
167     DefinitionModel model = new DefinitionModel(te, verified == null ? false : verified.root());
168 
169     errorIfInnerClass(te, messager);
170     
171     model.addDependencyNames(getImportsTypes(te));
172     String[] componentBeanNames  = AnnotationValueExtractor.getAnnotationValue(te, COMPONENT_TYPE, DEFAULT_ANNOTATION_VALUE);
173     if (componentBeanNames != null) {
174       errorOnBannedTypeToMessage(te, messager, COMPONENT_BANNED_ANNOTATIONS);
175       addModelsFromComponent(te, model, componentBeanNames, messager);
176     } else {
177       errorOnBannedTypeToMessage(te, messager, BEAN_LITE_BANNED_ANNOTATIONS);
178       for (Element enclosed : te.getEnclosedElements()) {
179         addBeanMethodsFromBeanLiteConfig(messager, model, enclosed);
180       }
181     }
182    
183     if (verified != null) {
184       for (String expectedBean : verified.expectedBeans()) {
185         model.addDefinition(new ExpectedModel(expectedBean, te));
186       }
187     }
188     return model;
189   }

191   private List<Modifier> getIllegalModifiers(Set<Modifier> existing, List<Modifier> illegal) {
192     List<Modifier> modifiers = new ArrayList<>(existing);
193     modifiers.removeIf(modifier -> !illegal.contains(modifier));
194     modifiers.sort((m1, m2) ->  m1.name().compareTo(m2.name())); //in case someone reorders
195     return modifiers;
196   }

198   private boolean parseBeanMethod(ExecutableElement beanMethod, String[] beanNames, Messager messager) {
199     boolean valid = true;
200     if (beanNames.length == 0) {
201       valid = false;
202       messager.printMessage(Kind.ERROR, "All @Bean annotations must define at least one name for a bean.", beanMethod);
203     }
204     if (beanMethod.getReturnType().getKind() != TypeKind.DECLARED) {
205       valid = false;
206       messager.printMessage(Kind.ERROR, "@Bean methods must return an Object", beanMethod);
207     }
208     if (!beanMethod.getModifiers().contains(Modifier.PUBLIC)) {
209       valid = false;
210       messager.printMessage(Kind.ERROR, "@Bean methods must be marked public", beanMethod);
211     }
212     List<Modifier> illegalModifiers = getIllegalModifiers(beanMethod.getModifiers(), DISALLOWED_ON_METHOD);
213     if (illegalModifiers.size() != 0) {
214       valid = false;
215       messager.printMessage(Kind.ERROR, "Illegal modifiers found on spring @Bean method: "
216            + illegalModifiers.stream().map(m -> m.name()).collect(Collectors.joining(", ")),
217           beanMethod);
218     }
219     return valid;
220   }

222   private void addBeanMethodsFromBeanLiteConfig(Messager messager, DefinitionModel model, Element enclosed) {
223     switch (enclosed.getKind()) {
224       case METHOD: 
225         ExecutableElement execelement = (ExecutableElement) enclosed;
226         String[] beanNames = AnnotationValueExtractor
227             .getAnnotationValue(execelement, "org.springframework.context.annotation.Bean", "name");
228         
229         if (beanNames != null) {
230           List<InstanceDependencyModel> dependencies = execElementDependency(messager, model, execelement);          
231           if (parseBeanMethod(execelement, beanNames, messager)) {
232             List<String> names = new ArrayList<>(Arrays.asList(beanNames));
233             String defaultName = names.get(0);
234             names.remove(defaultName);
235             model.addDefinition(new InstanceModel(defaultName, model.getIdentity(), execelement,
236                 execelement.getReturnType().toString(), dependencies, names));
237           }
238         } else {
239           messager.printMessage(Kind.ERROR, "All methods on @Configuration must have @Bean annotation", execelement);
240         }
241         break;
242       case FIELD:
243         if (!staticPrivateFinalLiteralField.test((VariableElement) enclosed)) {
244           messager.printMessage(Kind.ERROR, "Only private static final constants are permitted in @Verified @Configuration classes",
245               enclosed);
246         }
247         break;
248       case ENUM_CONSTANT: 
249         if (!staticPrivateFinalLiteralField.test((VariableElement) enclosed)) {
250           messager.printMessage(Kind.ERROR, "Only private static final constants are permitted in @Verified @Configuration classes",
251               enclosed);
252         }
253         break;
254       case CONSTRUCTOR:
255         ExecutableElement constelement = (ExecutableElement) enclosed;
256         if (!constelement.getModifiers().contains(Modifier.PUBLIC)) {
257           messager.printMessage(Kind.ERROR, "@Configuration should not have any non-public constructors.", enclosed);
258         }
259         if (constelement.getParameters().size() > 0) {
260           messager.printMessage(Kind.ERROR, "@Configuration should not have any non-default constructors.", enclosed);
261         }
262         break;
263       default:
264         messager.printMessage(Kind.ERROR, "Only @Bean methods, private static final literals, and default constructors "
265             + "are allowed on @Configuration classes", enclosed);
266         break;
267     }
268   }

270   /**
271    * This method is called on {@link ExecutableElement}.
272    * Bean methods on an @Configuration bean, or Constructors on @Component classes.
273    * This method parses the @Qualifier, or @Value annotations if a @Verified=root, and reads the types of each parameter.
274    * That data is used to build a list of {@link InstanceDependencyModel}'s which are part of an {@link InstanceModel}.
275    * All parameters must have an @Qualifier or @Value, and the annotations can not be mixed, errors will result otherwise.
276    * 
277    * @param messager APT messager that will receive error messages.
278    * @param model the DefinitionModel being parse, which may be a @Configuration or @Component annotated entity.
279    * @param execelement the bean method if an @Configuration, or the constructor if an @Component.
280    * @return the dependencies of the to be constructed {@link InstanceModel}
281    */
282   private List<InstanceDependencyModel> execElementDependency(Messager messager, DefinitionModel model,
283       ExecutableElement execelement) {
284     List<InstanceDependencyModel> dependencies = new ArrayList<>();
285     boolean hasValues = false;
286     boolean hasQualifiers = false;
287     for (VariableElement varelement : execelement.getParameters()) {
288       
289       String[] qualifierNames = AnnotationValueExtractor
290           .getAnnotationValue(varelement, QUALIFIER_TYPE, DEFAULT_ANNOTATION_VALUE);
291       String[] valueNames = AnnotationValueExtractor
292           .getAnnotationValue(varelement, VALUE_TYPE, DEFAULT_ANNOTATION_VALUE);
293       
294       if ((qualifierNames == null || qualifierNames.length == 0) 
295           && (valueNames == null || valueNames.length == 0)) {
296         messager.printMessage(Kind.ERROR, "All parameters must have an @Qualifier or a @Value annotation with a value", varelement);
297       } 
298       if (qualifierNames != null && qualifierNames.length > 0) {
299         dependencies.add(new InstanceDependencyModel(qualifierNames[0], varelement.asType().toString()));
300         hasQualifiers = true;
301       }
302       if (valueNames != null && valueNames.length > 0) {
303         //ignore values as they will be used to build beans and pass the data on, and
304         //are not beans themselves... and cannot be intermingled with @Qualifiers.
305         hasValues = true;
306       }
307     }
308     if (hasValues && hasQualifiers) {
309       messager.printMessage(Kind.ERROR, "No method may define both @Qualifier or a @Value annotations,"
310           + " keep property values in there own beans", execelement);
311     }
312     if (hasValues &&  !model.isRootNode()) {
313       messager.printMessage(Kind.ERROR, "Only @Verified(root=true) nodes may use @Value annotations to create beans,"
314           + " decouples spring graph from environment", execelement);
315     }
316     return dependencies;
317   }

319   /**
320    * Builds an instance model by finding the autowired constructor of an @Component model.
321    * 
322    * @param te the TypeElement corresponding to the @Component class.
323    * @param dm the definitionModel built from the @Component.
324    * @param names the list if names in an @Component annotation.  Users must explicitly define one.
325    * @param messager errors are added to this APT messager.
326    */
327   private void addModelsFromComponent(TypeElement te, DefinitionModel dm, String[] names, Messager messager) {
328     List<InstanceDependencyModel> dependencies = new ArrayList<>();
329     ExecutableElement chosenConstructor = findAutowiredConstructor(extractConstructorsFromComponent(te));
330     if (chosenConstructor == null) {
331       messager.printMessage(Kind.ERROR, "No single default constructor or single @Autowired constructor", te);
332     } else {
333       dependencies = execElementDependency(messager, dm, chosenConstructor);
334       //dm.getExpectedDefinitions()
335     }
336     te.getEnclosedElements().stream()
337         .filter(el -> el instanceof VariableElement)
338         .map(el -> (VariableElement) el)
339         .filter(ve -> !staticPrivateFinalLiteralField.test(ve) && !privateFinalField.test(ve))
340         .forEach(ve -> messager
341             .printMessage(Kind.ERROR, "@Component classes my only have static final constant fields or final private fields", ve));
342     
343     if (names.length > 0) {
344       InstanceModel model = new InstanceModel(names[0],
345           dm.getIdentity(), 
346           chosenConstructor, 
347           te.getQualifiedName().toString(),
348           dependencies, 
349           new ArrayList<>());
350       
351       dm.addDefinition(model);
352       for (InstanceDependencyModel dep : dependencies) {
353         ExpectedModel expectedModel = new ExpectedModel(dep.getIdentity());
354         expectedModel.addDefinitionReferenceToType(model.getIdentity(), dep.getType());
355         dm.addDefinition(expectedModel);
356       }
357     } else {
358       messager.printMessage(Kind.ERROR, "@Component classes must have a name", te);
359     }
360   }

362   /**
363    * Analyzes a list of constructors from an @Component, looking for a single constructor, or if multiple
364    * constructors exist, a single constructor marked with @Autowire.
365    *
366    * @param constructors a list of constructors from an @Component.
367    * @return the executable element, or null.
368    */
369   private ExecutableElement findAutowiredConstructor(List<ExecutableElement> constructors) {
370     ExecutableElement chosenConstructor = null;
371     if (constructors.size() == 1) {
372       chosenConstructor = constructors.get(0);
373     } else {
374       chosenConstructor = constructors.stream()
375         .filter(ex -> AnnotationValueExtractor.getAnnotationValue(ex, AUTOWIRED_TYPE, "") != null)
376         .limit(2) //stop at two. efficiency.
377         .reduce((a, b) -> null) //if more than one return null.
378         .orElse(null);
379     }
380     return chosenConstructor;
381   }

383   /**
384    * Given an @Component's {@link TypeElement} find's all constructors of that type.
385    * 
386    * @param te a representation of an @Component class.
387    * @return a list of executable elements representing all found constructors.
388    */
389   private List<ExecutableElement> extractConstructorsFromComponent(TypeElement te) {
390     return te.getEnclosedElements().stream()
391       .filter(enclosed -> enclosed instanceof ExecutableElement)
392       .filter(enclosed -> "<init>".equals(enclosed.getSimpleName().toString()))
393       .map(enclosed -> (ExecutableElement) enclosed)
394       .collect(Collectors.toList());
395   }

397   private void errorIfInnerClass(TypeElement te, Messager messager) {
398     if (te.getEnclosingElement().getKind() != ElementKind.PACKAGE && !isTest(te)) {
399       messager.printMessage(Kind.ERROR, "The class must be a top level class, not an internal class", te);
400     }
401   }

403   /**
404    * Returns true if the verified class is in a class that maven
405    * would consider a test.   May give false positives in class dir.
406    * TODO: should probably have a property to say if in tests.
407    * @param te the enclosed type
408    * @return true if determined to be in a test.
409    */
410   private boolean isTest(TypeElement te) {
411     String className = te.getEnclosingElement().getSimpleName().toString();
412     return className.startsWith("Test") 
413         || className.endsWith("Test") 
414         || className.endsWith("Tests")
415         || className.endsWith("TestCase");
416   }

418   private void errorOnBannedTypeToMessage(Element el, Messager messager, Map<String, String> typeToMessage) {
419     for (Entry<String, String> banned : typeToMessage.entrySet()) {
420       if (AnnotationValueExtractor.getAnnotationValue(el, banned.getKey(), "") != null) {
421         messager.printMessage(Kind.ERROR, banned.getValue(), el);
422       }
423     } 
424   }

426   private List<String> getImportsTypes(TypeElement element) {
427     String[] values = AnnotationValueExtractor
428         .getAnnotationValue(element, IMPORT_TYPE, DEFAULT_ANNOTATION_VALUE);
429     if (values == null) {
430       return new ArrayList<>();
431     } else {
432       return Arrays.asList(values);
433     }
434   }
